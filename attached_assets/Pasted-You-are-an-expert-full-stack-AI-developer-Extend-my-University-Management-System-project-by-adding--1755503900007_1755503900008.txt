🔹 Database Design
CREATE TABLE classrooms (
    id INT AUTO_INCREMENT PRIMARY KEY,
    room_no VARCHAR(20) UNIQUE
);

CREATE TABLE timetable (
    id INT AUTO_INCREMENT PRIMARY KEY,
    room_id INT,
    course_name VARCHAR(100),
    faculty_name VARCHAR(100),
    day_of_week ENUM('Mon','Tue','Wed','Thu','Fri','Sat'),
    start_time TIME,
    end_time TIME,
    FOREIGN KEY (room_id) REFERENCES classrooms(id)
);

GET /student/free-classrooms?day=Mon&start=10:00&end=11:00 → Return list of free rooms.

Free Classroom Logic

def get_free_classrooms(day, start_time, end_time):
    all_rooms = get_all_classrooms()
    occupied = db.query("""
        SELECT room_id FROM timetable
        WHERE day_of_week = %s 
        AND (start_time < %s AND end_time > %s)
    """, (day, end_time, start_time))
    
    free_rooms = set(all_rooms) - set([r['room_id'] for r in occupied])
    return free_rooms
